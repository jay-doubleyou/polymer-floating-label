floating input label with polymer

Usage

      <!DOCTYPE html>
      <html>
          <head>
              <script src="platform/platform.js"></script>
              <link rel="import" href="polymer/polymer.html">
              <link rel="import" href="floating-label/floating-label.html">
          </head>
          <body>
              <floating-label label="YOUR LABEL"></floating-label>
          </body>
      </html>
